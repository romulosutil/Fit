# FitOS MVP — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the FitOS MVP — a SaaS platform where Personal Trainers prescribe training and diet plans, and students track daily execution on mobile.

**Architecture:** Expo (React Native) monorepo for the mobile app; Supabase handles all backend concerns (auth, PostgreSQL, file storage, Row Level Security). No custom backend server — all business logic lives in the app and Supabase Edge Functions. Tabela TACO bundled as static JSON for the food database.

**Tech Stack:** Expo SDK 51 · React Native (TypeScript) · Supabase (Auth + PostgreSQL + Storage) · NativeWind v4 (Tailwind styling) · React Query (server state) · Zustand (local state) · Tabela TACO JSON

---

## Stack Decision Record (ADR-001)

| Concern | Choice | Reason |
|---|---|---|
| Mobile | Expo / React Native | Cross-platform, fastest MVP, OTA updates |
| Auth | Supabase Auth | Roles, JWT, magic link invite flow built-in |
| Database | Supabase (PostgreSQL) | Row Level Security = built-in personal/aluno isolation |
| Storage | Supabase Storage | Photo uploads (avaliação física) |
| Styling | NativeWind v4 | Tailwind tokens on mobile, Mobile First 375px |
| Food DB | TACO JSON (bundled) | No API dependency, offline-capable |
| State | React Query + Zustand | Server state separated from UI state |

`supabase/migrations/002_rls.sql`:
```sql
-- Habilitar RLS em todas as tabelas
alter table profiles enable row level security;
alter table personal_alunos enable row level security;
alter table anamneses enable row level security;
alter table avaliacoes_fisicas enable row level security;
alter table perimetros enable row level security;
alter table treinos enable row level security;
alter table exercicios_treino enable row level security;
alter table dietas enable row level security;
alter table refeicoes enable row level security;
alter table itens_refeicao enable row level security;
alter table logs_diarios enable row level security;
alter table checks_exercicio enable row level security;
alter table checks_refeicao enable row level security;
alter table streaks enable row level security;

-- profiles: cada um lê o próprio, personal lê os seus alunos
create policy "profiles_self" on profiles for select
  using (auth.uid() = id);

create policy "profiles_personal_reads_alunos" on profiles for select
  using (
    exists (
      select 1 from personal_alunos
      where personal_id = auth.uid() and aluno_id = profiles.id
    )
  );

-- personal_alunos: personal gerencia seus próprios vínculos
create policy "pa_personal_all" on personal_alunos for all
  using (personal_id = auth.uid());

create policy "pa_aluno_read" on personal_alunos for select
  using (aluno_id = auth.uid());

-- Helper: verifica se o caller é personal do aluno
create or replace function is_personal_of(p_aluno_id uuid)
returns boolean language sql security definer as $$
  select exists (
    select 1 from personal_alunos
    where personal_id = auth.uid() and aluno_id = p_aluno_id
  );
$$;

-- anamneses: aluno escreve o próprio, personal do aluno lê
create policy "anamnese_aluno_write" on anamneses for all
  using (aluno_id = auth.uid());

create policy "anamnese_personal_read" on anamneses for select
  using (is_personal_of(aluno_id));

-- avaliacoes_fisicas
create policy "aval_aluno_read" on avaliacoes_fisicas for select
  using (aluno_id = auth.uid());

create policy "aval_personal_all" on avaliacoes_fisicas for all
  using (personal_id = auth.uid() or is_personal_of(aluno_id));

-- treinos
create policy "treino_personal_all" on treinos for all
  using (personal_id = auth.uid());

create policy "treino_aluno_read" on treinos for select
  using (aluno_id = auth.uid());

-- dietas
create policy "dieta_personal_all" on dietas for all
  using (personal_id = auth.uid());

create policy "dieta_aluno_read" on dietas for select
  using (aluno_id = auth.uid());

-- logs_diarios, checks: aluno gerencia o próprio
create policy "logs_aluno" on logs_diarios for all using (aluno_id = auth.uid());
create policy "checks_exercicio_aluno" on checks_exercicio for all using (aluno_id = auth.uid());
create policy "checks_refeicao_aluno" on checks_refeicao for all using (aluno_id = auth.uid());
create policy "streaks_aluno" on streaks for all using (aluno_id = auth.uid());

-- personal lê logs dos seus alunos
create policy "logs_personal_read" on logs_diarios for select
  using (is_personal_of(aluno_id));
```

```typescript
import type { User } from "@supabase/supabase-js";

type Role = "personal" | "aluno" | null;

interface SessionStore {
  user: User | null;
  role: Role;
  setSession: (user: User | null, role: Role) => void;
  clearSession: () => void;
}

export const useSession = create<SessionStore>((set) => ({
  user: null,
  role: null,
  setSession: (user, role) => set({ user, role }),
  clearSession: () => set({ user: null, role: null }),
}));
```

`apps/mobile/lib/auth.ts`:
```typescript
import { supabase } from "./supabase";
import { useSession } from "../store/useSession";

export async function signIn(email: string, password: string) {
  const { data, error } = await supabase.auth.signInWithPassword({ email, password });
  if (error) throw error;
  const profile = await fetchProfile(data.user.id);
  useSession.getState().setSession(data.user, profile.role);
  return profile;
}

export async function registerPersonal(nome: string, email: string, password: string) {
  const { data, error } = await supabase.auth.signUp({ email, password });
  if (error) throw error;
  await supabase.from("profiles").insert({ id: data.user!.id, nome, email, role: "personal" });
  useSession.getState().setSession(data.user!, "personal");
}

export async function registerAluno(nome: string, email: string, password: string, personalId: string) {
  const { data, error } = await supabase.auth.signUp({ email, password });
  if (error) throw error;
  await supabase.from("profiles").insert({ id: data.user!.id, nome, email, role: "aluno" });
  await supabase.from("personal_alunos").insert({ personal_id: personalId, aluno_id: data.user!.id });
  useSession.getState().setSession(data.user!, "aluno");
}

export async function signOut() {
  await supabase.auth.signOut();
  useSession.getState().clearSession();
}

export async function fetchProfile(userId: string) {
  const { data, error } = await supabase.from("profiles").select("*").eq("id", userId).single();
  if (error) throw error;
  return data as { id: string; nome: string; email: string; role: "personal" | "aluno" };
}

export async function restoreSession() {
  const { data } = await supabase.auth.getSession();
  if (!data.session) return;
  const profile = await fetchProfile(data.session.user.id);
  useSession.getState().setSession(data.session.user, profile.role);
}
```

```typescript
import { useQuery, useMutation, useQueryClient } from "@tanstack/react-query";
import { supabase } from "../supabase";

interface AvaliacaoInput {
  aluno_id: string;
  personal_id: string;
  peso_kg?: number;
  altura_cm?: number;
  bf_pct?: number;
}

interface Avaliacao extends AvaliacaoInput {
  id: string;
  data: string;
  perimetros?: Perimetro;
}

interface Perimetro {
  braco_cm?: number;
  antebraco_cm?: number;
  peitoral_cm?: number;
  cintura_cm?: number;
  quadril_cm?: number;
  coxa_cm?: number;
  panturrilha_cm?: number;
}

export function useAvaliacoes(alunoId: string) {
  return useQuery({
    queryKey: ["avaliacoes", alunoId],
    queryFn: async () => {
      const { data, error } = await supabase
        .from("avaliacoes_fisicas")
        .select("*, perimetros(*)")
        .eq("aluno_id", alunoId)
        .order("data", { ascending: false });
      if (error) throw error;
      return data as Avaliacao[];
    },
    enabled: !!alunoId,
  });
}

export function useSalvarAvaliacao() {
  const qc = useQueryClient();
  return useMutation({
    mutationFn: async ({
      avaliacao,
      perimetros,
    }: {
      avaliacao: AvaliacaoInput;
      perimetros: Perimetro;
    }) => {
      const { data, error } = await supabase
        .from("avaliacoes_fisicas")
        .insert(avaliacao)
        .select()
        .single();
      if (error) throw error;
      await supabase.from("perimetros").insert({ avaliacao_id: data.id, ...perimetros });
      return data;
    },
    onSuccess: (_data, variables) =>
      qc.invalidateQueries({ queryKey: ["avaliacoes", variables.avaliacao.aluno_id] }),
  });
}
```

```typescript
      personalId: string;
      nome: string;
```

```typescript
import { useQuery, useMutation, useQueryClient } from "@tanstack/react-query";
import { supabase } from "../supabase";
import tacoData from "../../assets/taco.json";

export type TacoItem = { id: number; nome: string; kcal: number; ptn_g: number; cho_g: number; lip_g: number };

export function searchTaco(query: string): TacoItem[] {
  if (query.length < 2) return [];
  return (tacoData as TacoItem[])
    .filter((item) => item.nome.toLowerCase().includes(query.toLowerCase()))
    .slice(0, 8);
}

export function calcTDEE(pesoKg: number, alturaCm: number, idadeAnos: number, sexo: "M" | "F", nivelAtividade: number): number {
  const tmb =
    sexo === "M"
      ? 10 * pesoKg + 6.25 * alturaCm - 5 * idadeAnos + 5
      : 10 * pesoKg + 6.25 * alturaCm - 5 * idadeAnos - 161;
  return Math.round(tmb * nivelAtividade);
}

export function useDieta(alunoId: string) {
  return useQuery({
    queryKey: ["dieta", alunoId],
    queryFn: async () => {
      const { data, error } = await supabase
        .from("dietas")
        .select("*, refeicoes(*, itens_refeicao(*))")
        .eq("aluno_id", alunoId)
        .eq("ativa", true)
        .maybeSingle();
      if (error) throw error;
      return data;
    },
    enabled: !!alunoId,
  });
}

export function useCriarDieta() {
  const qc = useQueryClient();
  return useMutation({
    mutationFn: async ({
      alunoId,
      personalId,
      calorias,
      proteina,
      carbo,
      gordura,
      refeicoes,
    }: {
      alunoId: string;
      personalId: string;
      calorias: number;
      proteina: number;
      carbo: number;
      gordura: number;
      refeicoes: Array<{
        nome: string;
        horario: string;
        ordem: number;
        itens: Array<{ alimento_nome: string; quantidade_g: number; calorias: number; proteina_g: number; carbo_g: number; gordura_g: number }>;
      }>;
    }) => {
      const { data: dieta, error } = await supabase
        .from("dietas")
        .insert({ aluno_id: alunoId, personal_id: personalId, calorias_meta: calorias, proteina_g: proteina, carbo_g: carbo, gordura_g: gordura })
        .select()
        .single();
      if (error) throw error;
      for (const ref of refeicoes) {
        const { data: r, error: rErr } = await supabase
          .from("refeicoes")
          .insert({ dieta_id: dieta.id, nome: ref.nome, horario: ref.horario, ordem: ref.ordem })
          .select()
          .single();
        if (rErr) throw rErr;
        if (ref.itens.length > 0) {
          const { error: iErr } = await supabase
            .from("itens_refeicao")
            .insert(ref.itens.map((i) => ({ ...i, refeicao_id: r.id })));
          if (iErr) throw iErr;
        }
      }
      return dieta;
    },
    onSuccess: (_data, vars) => qc.invalidateQueries({ queryKey: ["dieta", vars.alunoId] }),
  });
}
```

```sql
create or replace function update_streak(p_aluno_id uuid)
returns void language plpgsql security definer as $$
declare
  v_ultima date;
  v_streak int;
begin
  select ultima_atividade, streak_atual into v_ultima, v_streak
  from streaks where aluno_id = p_aluno_id;

  if not found then
    insert into streaks(aluno_id, streak_atual, streak_max, xp_total, ultima_atividade)
    values (p_aluno_id, 1, 1, 0, current_date);
    return;
  end if;

  if v_ultima = current_date then return; end if;

  if v_ultima = current_date - 1 then
    v_streak := v_streak + 1;
  else
    v_streak := 1;
  end if;

  update streaks set
    streak_atual = v_streak,
    streak_max = greatest(streak_max, v_streak),
    ultima_atividade = current_date
  where aluno_id = p_aluno_id;
end;
$$;

-- Atualiza XP total quando log é salvo
create or replace function sync_xp()
returns trigger language plpgsql security definer as $$
begin
  update streaks
  set xp_total = xp_total + new.xp_ganho
  where aluno_id = new.aluno_id;
  return new;
end;
$$;
```

```typescript
export async function uploadFotoAvaliacao(
  alunoId: string,
  posicao: "frente" | "costas" | "lado",
  uri: string
): Promise<string> {
  const ext = uri.split(".").pop() ?? "jpg";
  const path = `avaliacoes/${alunoId}/${Date.now()}_${posicao}.${ext}`;
  const response = await fetch(uri);
  const blob = await response.blob();
  const { error } = await supabase.storage.from("fotos-avaliacao").upload(path, blob, {
    contentType: `image/${ext}`,
    upsert: false,
  });
  if (error) throw error;
  const { data } = supabase.storage.from("fotos-avaliacao").getPublicUrl(path);
  return data.publicUrl;
}

  const perm = await ImagePicker.requestMediaLibraryPermissionsAsync();
  if (!perm.granted) return null;
  const result = await ImagePicker.launchImageLibraryAsync({ mediaTypes: ImagePicker.MediaTypeOptions.Images, quality: 0.7 });
  if (result.canceled) return null;
  return uploadFotoAvaliacao(alunoId, posicao, result.assets[0].uri);
}
```
